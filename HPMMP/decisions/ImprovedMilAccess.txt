political_decisions = {
    give_military_access = {
        potential = {
            OR= {
    	    	AND = { 
    	    		is_vassal = yes
    	    		overlord = { NOT = { military_access = THIS } }
    	    	}
    	    	AND = { 	
    	    		part_of_sphere = yes
    	    		sphere_owner = { NOT = { military_access = THIS } }
    	    	}
        	}
        }

        allow = {
        	OR= {
    	    	AND = { 
    	    		is_vassal = yes
    	    		overlord = { NOT = { military_access = THIS } }
    	    	}
    	    	AND = { 	
    	    		part_of_sphere = yes
    	    		sphere_owner = { NOT = { military_access = THIS } }
    	    	}
        	}
        }

        effect = {
        	any_country = {
        		limit = {
        			exists = yes
                    NOT = { tag = THIS }
        			NOT = { military_access = THIS }
                    OR = {
                        is_our_vassal = THIS
                        is_sphere_leader_of = THIS
                    }
        		}
        		military_access = THIS
        	}
        }

        ai_will_do = {
        	factor = 1
        }
    }

    remove_military_access = {
    	potential = {
            OR = {
                any_greater_power = {
             		military_access = THIS
             		NOT = {
             			OR = {
        		 			is_our_vassal = THIS
        	     			is_sphere_leader_of = THIS
             			}
             		}
        	    }
                any_neighbor_country = { 
                    military_access = THIS
                    NOT = {
                        OR = {
                            is_our_vassal = THIS
                            is_sphere_leader_of = THIS
                        }
                    }
                }
            }
        }

        allow = {
        	OR = {
                any_greater_power = {
                    military_access = THIS
                    NOT = {
                        OR = {
                            is_our_vassal = THIS
                            is_sphere_leader_of = THIS
                        }
                    }
                }
                any_neighbor_country = { 
                    military_access = THIS
                    NOT = {
                        OR = {
                            is_our_vassal = THIS
                            is_sphere_leader_of = THIS
                        }
                    }
                }
            }
        }

        effect = {
        	any_country = {
        		limit = { 
                    exists = yes
        			military_access = THIS
        			NOT = {
    	    			is_sphere_leader_of = THIS
    	    			is_our_vassal = THIS
    	    		}
        		}
        		end_military_access = THIS
        	}
        }

        ai_will_do = {
        	factor = 1
        }
    }
}
